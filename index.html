Проектирование структуры БД

🔗 DDL-схема сформирована на основе полей прототипа с учетом физического распределения данных (NVMe/SATA).

📜 DDL-скрипт (PostgreSQL)

-- Таблица абонементов (секционирована по месяцам)
CREATE TABLE passes (
    id UUID PRIMARY KEY,
    client_name VARCHAR(150),
    section_name VARCHAR(100),
    pass_type VARCHAR(50),
    valid_from DATE,
    valid_to DATE,
    status VARCHAR(20) DEFAULT 'ACTIVE'
) PARTITION BY RANGE (valid_from);

-- Таблица тренировок (секционирована по датам)
CREATE TABLE trainings (
    id UUID PRIMARY KEY,
    training_date DATE,
    training_time TIME,
    section_name VARCHAR(100),
    hall_name VARCHAR(50),
    coach_name VARCHAR(150),
    status VARCHAR(20) DEFAULT 'DRAFT'
) PARTITION BY RANGE (training_date);

-- Таблица тренеров с учетом нагрузки
CREATE TABLE coaches (
    id UUID PRIMARY KEY,
    full_name VARCHAR(150),
    specialization VARCHAR(100),
    qualification VARCHAR(50),
    monthly_hours INT DEFAULT 0,
    current_load DECIMAL(5,2) DEFAULT 0
);
